# there's no way for us to easily catch every character that's generated
# so instead we have to do this horribly clunky workaround...

# "zzz" so that this on_game_start_after_lobby fires last after all pdx events (hopefully)

on_game_start_after_lobby = {
    on_actions = {
        hcg_height_character_initialization
    }
}

hcg_height_character_initialization = {
    trigger = {
        has_game_rule = hcg_height_enabled_true
    }

    effect = {
        every_living_character = {
            hcg_seed_height_traits_effect = yes
        }
    }
}

hcg_character_initialization_pulse = {
    trigger = {
        has_game_rule = hcg_height_enabled_true
        NOR = { 
            has_character_flag = hcg_height_traits_seeded
            hcg_has_any_height_gene_trigger = yes
        }
    }
	effect = {
        hcg_seed_height_traits_effect = yes
    }
}

on_join_court = {
	on_actions = { hcg_character_initialization_pulse }
}

on_birthday = {
	on_actions = { hcg_character_initialization_pulse }
}

on_title_gain = {
	on_actions = { hcg_character_initialization_pulse }
}

on_birth_child = {
	on_actions = { hcg_character_initialization_pulse }
}
